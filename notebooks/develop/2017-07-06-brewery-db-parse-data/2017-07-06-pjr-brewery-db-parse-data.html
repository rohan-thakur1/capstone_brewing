# encode strings with utf-8
def encode_str(x):
    return str(x).encode('utf-8')

# encode all string columns
def clean_row(x):
    for c in x.keys():
        if isinstance(c, str):
            x[c] = encode_str(x[c])
    return x

def append_record(data, df, id='id'):
    if not data[id] in df[id].tolist():
        df.append(data, ignore_index=True)
    return

def get_object(r, df):
    data = pd.DataFrame()
    for c in df.columns:
        data[c] = r[c]
    return data

def get_beer(r):
    # build beer object
    beer = get_object(r, beers)

    append_record(beer, beers)

#define worker
def parse_data(i, q):
    while True:
        # get json 
        page_number = q.get()
        fpath = os.path.abspath(os.path.join(data_dir, '{}.json'.format(page_number)))
        page = pd.read_json(fpath)

        #get beer data from json
        records = page['data']
        
        # iterate over records
        for r in records:
            get_beer(r)
            get_srm(r)
            get_glass(r)
            get_style(r)
            get_breweries(r)
        
        return
